import java.util.Scanner;

public class Main_15651 {
	static int N, M;
	static int[] arr;
	static StringBuilder sb = new StringBuilder();
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		N = sc.nextInt();
		M = sc.nextInt();
		
		// 길이가 M인 수열을 담을 배열
		arr = new int[M];
		
		backTracking(0);
		System.out.print(sb);
	}
	
	static void backTracking(int depth) {
		
		// 수열의 길이가 M과 같다면 출력
		if(M == depth) {
			for(int i : arr) {
				sb.append(i + " ");
			}
			sb.append("\n");
			return;
		}
	
		for(int i = 1; i <= N; i++) {
			arr[depth] = i;
			backTracking(depth+1);
		}
	}
}
